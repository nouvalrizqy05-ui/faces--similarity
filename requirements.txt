# ─── Session State ────────────────────────────────────────────
def init_state():
    defaults = {
        "pca"    : None,
        "X"      : None,
        "Z"      : None,
        "labels" : None,
        "paths"  : None,
        "trained": False,
        "dataset_ready": False,
        "dataset_path" : "dataset/",
    }
    for k, v in defaults.items():
        if k not in st.session_state:
            st.session_state[k] = v

init_state()


# ─── Auto-load model dari models/ jika belum dilatih ─────
def try_autoload_model():
    """
    Coba load model dari models/ secara otomatis saat app pertama dibuka.
    Ini memungkinkan app langsung bisa dipakai tanpa perlu klik Training.
    """
    if st.session_state.trained:
        return  # sudah ter-load, skip

    # Cari model di beberapa lokasi yang mungkin
    base_dir = os.path.dirname(os.path.dirname(os.path.abspath(__file__)))
    candidate_dirs = [
        os.path.join(base_dir, "models"),           # root/models/
        os.path.join(os.path.dirname(__file__), "..", "models"),  # relative
        "models",                                    # cwd
    ]

    model_path = None
    db_path    = None

    for d in candidate_dirs:
        mp = os.path.join(d, "pca_model.pkl")
        dp = os.path.join(d, "face_database.pkl")
        if os.path.exists(mp) and os.path.exists(dp):
            model_path = mp
            db_path    = dp
            break

    if model_path is None:
        return  # tidak ada model, biarkan user training manual

    try:
        pca = FacePCA()
        pca.load(model_path)

        with open(db_path, "rb") as f:
            db = pickle.load(f)

        st.session_state.pca     = pca
        st.session_state.Z       = db["Z"]
        st.session_state.labels  = db["labels"]
        st.session_state.paths   = db.get("paths", [])
        st.session_state.trained = True

        # Coba load X juga kalau ada
        X_path = os.path.join(os.path.dirname(model_path), "X_data.pkl")
        if os.path.exists(X_path):
            with open(X_path, "rb") as f:
                st.session_state.X = pickle.load(f)
        else:
            # Rekonstruksi X dari Z dan model (approx)
            Z   = db["Z"]
            Xc_approx = Z @ pca.components_
            st.session_state.X = Xc_approx + pca.mean_face_

    except Exception:
        pass  # gagal load, biarkan user training manual


try_autoload_model()

def fig_to_bytes(fig) -> bytes:
    buf = io.BytesIO()
    fig.savefig(buf, format="png", dpi=150, bbox_inches="tight")
    buf.seek(0)
    return buf.read()


def metric_card(label: str, value: str):
    st.markdown(f"""
    <div class="metric-card">
        <div class="label">{label}</div>
        <div class="value">{value}</div>
    </div>""", unsafe_allow_html=True)


def img_from_upload(upload) -> np.ndarray:
    """Baca gambar dari UploadedFile → ndarray float32 grayscale [0,1]."""
    import cv2
    file_bytes = np.frombuffer(upload.read(), np.uint8)
    img = cv2.imdecode(file_bytes, cv2.IMREAD_COLOR)
    gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
    resized = cv2.resize(gray, IMG_SIZE)
    return resized.astype(np.float32) / 255.0


def vec_from_upload(upload) -> np.ndarray:
    return img_from_upload(upload).flatten()
